2> Adicionar esse campo em App.config//Web.Config

  <connectionStrings>
    <add name="Nome do banco" connectionString="data source=.\SQLEXPRESS;Integrated Security=SSPI;Initial Catalog=Nome do banco"
         providerName="System.Data.SqlClient"/>
  </connectionStrings>

  // Esse é para bancos internos da aplicação, por enquanto não vou usar nada interno, mas vou deixar aqui para o futuro
